export type ObjectDataCollection = ObjectData[];

/**
 * Data about how the object should be displayed
 */
export interface DisplayData {
    /**
     * The altitude of the object to be placed
     */
    alt: number;

    /**
     * The latitude of the object to be placed
     */
    lat: number;

    /**
     * The longitude of the object to be placed
     */
    lng: number;

    /**
     * The type of object
     */
    type: DisplayType;
}

/**
 * The Type of the Object to be displayed
 */
export enum DisplayType {
    /**
     * The object should be displayed as an earth satellite
     */
    EarthSatellite = "earth-satellite",

    /**
     * The object should be displayed as an earth station
     */
    EarthStation = "earth-station",

    /**
     * The object should be displayed as a geostationary satellite
     */
    GeoSatellite = "geo-satellite",
}

export interface ObjectData {
    /**
     * The displaydata of the object
     */
    display: DisplayData;

    /**
     * All information about the object
     */
    information: ObjectInformation[];
}

/**
 * Single information about how the object
 */
export interface ObjectInformation {
    /**
     * The information object; TODO: Create object types as specific classes? /
     * otherwise somehow check this
     */
    information: any;

    /**
     * The type of information this is
     */
    type: ObjectInformationType;

    /**
     * Weather the information should be displayed in the detail view
     */
    detail?: boolean;

    /**
     * Weather the information should be displayed in the hover view
     */
    hover?: boolean;
}

/**
 * The type of Information
 */
export enum ObjectInformationType {
    /**
     * Information about the objects altitude
     */
    Altitude = "altitude",

    /**
     * The objects description
     */
    Description = "description",
}
