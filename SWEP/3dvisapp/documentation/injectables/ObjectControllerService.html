import { Injectable } from "@angular/core";
import * as THREE from "three";
import { ObjectGeneratorService } from "@service/object-generator/object-generator.service";
import { ObjectDataService } from "@service/object-data/object-data.service";
import { Earth } from "@model/earth/earth.model";
import { PlaceableObject } from "@model/placeable-object/placeable-object.model";
import { DisplayType, ObjectDataCollection } from "@service/object-data/object-data-collection.interface";

@Injectable({
  providedIn: "root"
})
export class ObjectControllerService {

  private earthObject: Earth;
  private orbit: THREE.Mesh;
  private earthStations: PlaceableObject[];
  private earthSatellites: PlaceableObject[];
  private geoSatellites: PlaceableObject[];

  constructor(private objectGenerator: ObjectGeneratorService, private objectDataService: ObjectDataService) {

    this.earthSatellites = new Array<PlaceableObject>();
    this.earthStations = new Array<PlaceableObject>();
    this.geoSatellites = new Array<PlaceableObject>();
    this.createInitialObjects();
    this.createPlaceables();

  }

  private createInitialObjects(): void {

    this.orbit = this.objectGenerator.createOrbit(50, 64, 64);
    this.earthObject = this.objectGenerator.createEarthObject(10, 64, 64);

  }

  private createPlaceables(): void {

    this.objectDataService.getObjectData().then(objectDataCollection => {

      for (let displayData of objectDataCollection.map(objectData => objectData.display)) {

        switch (displayData.type) {

          case DisplayType.EarthStation:
            this.earthStations.push(this.objectGenerator.createEarthStation(displayData));
            break;
          case DisplayType.EarthSatellite:
            this.earthSatellites.push(this.objectGenerator.createEarthSatellitie(displayData));
            break;
          case DisplayType.GeoSatellite:
            this.geoSatellites.push(this.objectGenerator.createGeoSatellite(displayData));
            break;
        }
      }
      this.placeEarthStations();
      this.placeEarthSatellites();
      this.placeGeoSatellites();
      //this.detachEarthStations();
      //this.earthObject.getSurfaceMesh().add(...this.earthStations.map(placeableObject => placeableObject.getMesh()));
      //this.earthObject.getSurfaceMesh().add(...this.earthSatellites.map(placeableObject => placeableObject.getMesh()));
      //this.earthObject.getSurfaceMesh().add(...this.geoSatellites.map(placeableObject => placeableObject.getMesh()));

    }).catch(error => console.log(error));

  }

  public scalePlaceables(camera: THREE.Camera): void {

    let scale = this.calcScale(camera);
    this.scale(scale);

  }

  private calcScale(camera: THREE.Camera): number {

    const factor = 30;
    let earthPos: THREE.Vector3 = this.earthObject.getGetMeshGroup().position;
    let cameraPos: THREE.Vector3 = camera.position;
    let scale: number = new THREE.Vector3().subVectors(earthPos, cameraPos).length() / factor;

    return scale;

  }

  private scale(scale: number): void {

    this.earthStations.map((x) => x.getMesh().scale.set(scale, scale, scale));
    this.earthSatellites.map((x) => x.getMesh().scale.set(scale, scale, scale));
    // this.geoSatellites.map((x) => x.getMesh().scale.set(scale, scale, scale));

  }

  public placeEarthStations() {
    this.earthObject.getSurfaceMesh().add(...this.earthStations.map(placeableObject => placeableObject.getMesh()));
  }

  public removeEarthStations() {
    this.earthObject.getSurfaceMesh().remove(...this.earthStations.map(placeableObject => placeableObject.getMesh()));
  }

  public placeEarthSatellites() {
    this.earthObject.getSurfaceMesh().add(...this.earthSatellites.map(placeableObject => placeableObject.getMesh()));
  }

  public removeEarthSatellites() {
    this.earthObject.getSurfaceMesh().remove(...this.earthSatellites.map(placeableObject => placeableObject.getMesh()));
  }

  public placeGeoSatellites() {
    this.earthObject.getSurfaceMesh().add(...this.geoSatellites.map(placeableObject => placeableObject.getMesh()));
  }

  public removeGeoSatellites() {
    this.earthObject.getSurfaceMesh().remove(...this.geoSatellites.map(placeableObject => placeableObject.getMesh()));
  }

  public getEarth(): Earth {

    return this.earthObject;

  }

  public getOribit(): THREE.Mesh {

    return this.orbit;

  }

  public getEarthStations(): PlaceableObject[] {

    return this.earthStations;

  }

  public getEarthSatellites(): PlaceableObject[] {

    return this.earthSatellites;

  }

  public getGeoSatellites(): PlaceableObject[] {

    return this.geoSatellites;

  }

}
